	      http://overpass-turbo.eu/
	      This has been generated by the overpass-turbo wizard.
	      The original search was:
	      “rail”
	      */
	      [out:json][timeout:25];
	      // gather results
	      (
	      // query part for: “rail”
	      way["railway"="rail"]["usage"="main"]({{bbox}});
	      );
	      // print results
	      out body;
	      >;
	      out skel qt;
